#cloud-config
package_update: true
package_upgrade: true
packages:
  - apt-transport-https
  - ca-certificates
  - curl
  - gnupg
  - lsb-release
  - conntrack
  - socat
  - conntrack
  - apt-transport-https
runcmd:
  # configure Docker repository and install Docker CE
  - curl -fsSL https://download.docker.com/linux/ubuntu/gpg | gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg
  - echo "deb [arch=amd64 signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable" | tee /etc/apt/sources.list.d/docker.list > /dev/null
  - apt-get update
  - apt-get install -y docker-ce docker-ce-cli containerd.io
  - systemctl enable --now docker

  # install kubectl
  - curl -LO "https://dl.k8s.io/release/$(curl -L -s https://dl.k8s.io/release/stable.txt)/bin/linux/amd64/kubectl"
  - install -o root -g root -m 0755 kubectl /usr/local/bin/kubectl

  # install minikube
  - curl -Lo minikube https://storage.googleapis.com/minikube/releases/latest/minikube-linux-amd64
  - install minikube /usr/local/bin/minikube

  # disable swap (required by kube)
  - swapoff -a
  - sed -i.bak '/ swap / s/^\(.*\)$/#\1/g' /etc/fstab || true

  # Add current admin user to docker group (cloud-init provides ubuntu or azureuser depending)
  - usermod -aG docker ${USER:-ubuntu}
  - usermod -aG docker ${USER:-azureuser}
  - systemctl restart docker || true

  # Wait a bit for docker to be ready
  - sleep 5

  # start minikube using the "none" driver (runs kubelet on the host), run as root
  - /usr/local/bin/minikube start --driver=none --kubernetes-version=v1.28.0 || true

  # allow kubectl to access cluster
  - mkdir -p /root/.kube
  - cp /etc/kubernetes/admin.conf /root/.kube/config || true

  # adjust permissions for non-root user
  - chown -R ${USER:-ubuntu}:${USER:-ubuntu} /root/.kube || true

  # enable addons commonly useful
  - /usr/local/bin/minikube addons enable metrics-server || true
  - /usr/local/bin/minikube status
